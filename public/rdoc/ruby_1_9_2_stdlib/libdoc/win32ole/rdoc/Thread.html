def initialize(*arg, &block)
  if block
    org_initialize(*arg) {
      WIN32OLE.ole_initialize
      begin
        block.call(*arg)
      ensure
        WIN32OLE.ole_uninitialize
      end
    }
  else
    org_initialize(*arg)
  end
end
